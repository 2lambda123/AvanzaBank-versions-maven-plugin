92      /**
93       * Returns a set of dependencies where the dependencies which are defined in the dependency management section have
94       * been filtered out.
95       *
96       * @param dependencies The set of dependencies.
97       * @param dependencyManagement The set of dependencies from the dependency management section.
98       * @return A new set of dependencies which are from the set of dependencies but not from the set of dependency
99       *         management dependencies.
100      * @since 1.0-beta-1
101      */
102     private static Set removeDependencyManagment( Set dependencies, Set dependencyManagement )
103     {
104         Set result = new TreeSet( new DependencyComparator() );
105         for ( Iterator i = dependencies.iterator(); i.hasNext(); )
106         {
107             Dependency c = (Dependency) i.next();
108             boolean matched = false;
109             Iterator j = dependencyManagement.iterator();
110             while ( !matched && j.hasNext() )
111             {
112                 Dependency t = (Dependency) j.next();
113                 if ( StringUtils.equals( t.getGroupId(), c.getGroupId() )
114                     && StringUtils.equals( t.getArtifactId(), c.getArtifactId() )
115                     && ( t.getScope() == null || StringUtils.equals( t.getScope(), c.getScope() ) )
116                     && ( t.getClassifier() == null || StringUtils.equals( t.getClassifier(), c.getClassifier() ) )
117                     && ( c.getVersion() == null || t.getVersion() == null
118                         || StringUtils.equals( t.getVersion(), c.getVersion() ) ) )
119                 {
120                     matched = true;
121                     break;
122                 }
123             }
124             if ( !matched )
125             {
126                 result.add( c );
127             }
128         }
129         return result;
130     }

154     /**
155      * @throws org.apache.maven.plugin.MojoExecutionException when things go wrong
156      * @throws org.apache.maven.plugin.MojoFailureException when things go wrong in a very bad way
157      * @see org.codehaus.mojo.versions.AbstractVersionsUpdaterMojo#execute()
158      * @since 1.0-alpha-1
159      */
160     public void execute()
161         throws MojoExecutionException, MojoFailureException
162     {
163         logInit();
164         Set dependencyManagement = new TreeSet( new DependencyComparator() );
165         dependencyManagement.addAll( getProject().getDependencyManagement() == null ? Collections.EMPTY_LIST
166                         : getProject().getDependencyManagement().getDependencies() );
167 
168         Set dependencies = new TreeSet( new DependencyComparator() );
169         dependencies.addAll( getProject().getDependencies() );
170         if ( isProcessingDependencyManagement() )
171         {
172             dependencies = removeDependencyManagment( dependencies, dependencyManagement );
173         }
174 
175         try
176         {
177             if ( isProcessingDependencyManagement() )
178             {
179                 logUpdates( getHelper().lookupDependenciesUpdates( dependencyManagement, false ),
180                             "Dependency Management" );
181             }
182             if ( isProcessingDependencies() )
183             {
184                 logUpdates( getHelper().lookupDependenciesUpdates( dependencies, false ), "Dependencies" );
185             }
186         }
187         catch ( InvalidVersionSpecificationException e )
188         {
189             throw new MojoExecutionException( e.getMessage(), e );
190         }
191         catch ( ArtifactMetadataRetrievalException e )
192         {
193             throw new MojoExecutionException( e.getMessage(), e );
194         }
195     }

197     private void logUpdates( Map updates, String section )
198     {
199         List withUpdates = new ArrayList();
200         List usingCurrent = new ArrayList();
201         Iterator i = updates.values().iterator();
202         while ( i.hasNext() )
203         {
204             ArtifactVersions versions = (ArtifactVersions) i.next();
205             String left = "  " + ArtifactUtils.versionlessKey( versions.getArtifact() ) + " ";
206             final String current;
207             ArtifactVersion latest;
208             if ( versions.isCurrentVersionDefined() )
209             {
210                 current = versions.getCurrentVersion().toString();
211                 latest = versions.getNewestUpdate( UpdateScope.ANY, Boolean.TRUE.equals( allowSnapshots ) );
212             }
213             else
214             {
215                 ArtifactVersion newestVersion = versions.getNewestVersion( versions.getArtifact().getVersionRange(),
216                                                                            Boolean.TRUE.equals( allowSnapshots ) );
217                 current = versions.getArtifact().getVersionRange().toString();
218                 latest = newestVersion == null ? null
219                                 : versions.getNewestUpdate( newestVersion, UpdateScope.ANY,
220                                                             Boolean.TRUE.equals( allowSnapshots ) );
221                 if ( latest != null
222                     && ArtifactVersions.isVersionInRange( latest, versions.getArtifact().getVersionRange() ) )
223                 {
224                     latest = null;
225                 }
226             }
227             String right = " " + ( latest == null ? current : current + " -> " + latest.toString() );
228             List t = latest == null ? usingCurrent : withUpdates;
229             if ( right.length() + left.length() + 3 > INFO_PAD_SIZE )
230             {
231                 t.add( left + "..." );
232                 t.add( StringUtils.leftPad( right, INFO_PAD_SIZE ) );
233 
234             }
235             else
236             {
237                 t.add( StringUtils.rightPad( left, INFO_PAD_SIZE - right.length(), "." ) + right );
238             }
239         }
240         if ( isVerbose() && usingCurrent.isEmpty() && !withUpdates.isEmpty() )
241         {
242             logLine( false, "No dependencies in " + section + " are using the newest version." );
243             logLine( false, "" );
244         }
245         else if ( isVerbose() && !usingCurrent.isEmpty() )
246         {
247             logLine( false, "The following dependencies in " + section + " are using the newest version:" );
248             i = usingCurrent.iterator();
249             while ( i.hasNext() )
250             {
251                 logLine( false, (String) i.next() );
252             }
253             logLine( false, "" );
254         }
255         if ( withUpdates.isEmpty() && !usingCurrent.isEmpty() )
256         {
257             logLine( false, "No dependencies in " + section + " have newer versions." );
258             logLine( false, "" );
259         }
260         else if ( !withUpdates.isEmpty() )
261         {
262             logLine( false, "The following dependencies in " + section + " have newer versions:" );
263             i = withUpdates.iterator();
264             while ( i.hasNext() )
265             {
266                 logLine( false, (String) i.next() );
267             }
268             logLine( false, "" );
269         }
270     }
